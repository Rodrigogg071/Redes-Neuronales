# Paso 0  Configuracin inicial y preparacin de datos (MNIST)

import numpy as np
import matplotlib.pyplot as plt
import tensorflow as tf
from tensorflow.keras.datasets import mnist
from tensorflow.keras.utils import to_categorical

# -------------------------------------------------------------------
# 1. Cargar el dataset MNIST
# -------------------------------------------------------------------
(x_train, y_train), (x_test, y_test) = mnist.load_data()

print("Shapes originales:")
print("x_train:", x_train.shape)
print("y_train:", y_train.shape)
print("x_test :", x_test.shape)
print("y_test :", y_test.shape)

# -------------------------------------------------------------------
# 2. Normalizar imgenes a [0, 1] y convertir a float32
# -------------------------------------------------------------------
x_train = x_train.astype("float32") / 255.0
x_test  = x_test.astype("float32") / 255.0

# -------------------------------------------------------------------
# 3. Aadir canal (para usar CNN: 28x28x1)
# -------------------------------------------------------------------
# x_train: (60000, 28, 28) -> (60000, 28, 28, 1)
# x_test : (10000, 28, 28) -> (10000, 28, 28, 1)
x_train = np.expand_dims(x_train, axis=-1)
x_test  = np.expand_dims(x_test, axis=-1)

print("\nShapes despus de aadir canal:")
print("x_train:", x_train.shape)
print("x_test :", x_test.shape)

# -------------------------------------------------------------------
# 4. Separar conjunto de validacin
#    Usaremos 50,000 para entrenamiento y 10,000 para validacin
# -------------------------------------------------------------------
x_val = x_train[50000:]
y_val = y_train[50000:]

x_train = x_train[:50000]
y_train = y_train[:50000]

print("\nShapes despus de separar validacin:")
print("x_train:", x_train.shape)
print("y_train:", y_train.shape)
print("x_val  :", x_val.shape)
print("y_val  :", y_val.shape)

# -------------------------------------------------------------------
# 5. One-hot encoding de las etiquetas (10 clases: 09)
# -------------------------------------------------------------------
num_classes = 10

y_train_cat = to_categorical(y_train, num_classes)
y_val_cat   = to_categorical(y_val, num_classes)
y_test_cat  = to_categorical(y_test, num_classes)

print("\nShapes de etiquetas codificadas (one-hot):")
print("y_train_cat:", y_train_cat.shape)
print("y_val_cat  :", y_val_cat.shape)
print("y_test_cat :", y_test_cat.shape)

# -------------------------------------------------------------------
# 6. Comprobacin rpida de rangos de pxeles
# -------------------------------------------------------------------
print("\nRango de valores de x_train (min, max):",
      float(x_train.min()), float(x_train.max()))
print("Rango de valores de x_test  (min, max):",
      float(x_test.min()), float(x_test.max()))

# Paso 1  Definicin y entrenamiento del modelo base (CNN sobre MNIST)

import tensorflow as tf
from tensorflow.keras import layers, models
import matplotlib.pyplot as plt

# Por si quieres fijar semilla para reproducibilidad (opcional):
tf.random.set_seed(42)

# -------------------------------------------------------------------
# 1. Definir el modelo base (CNN)
# -------------------------------------------------------------------
model_base = models.Sequential(
    [
        layers.Input(shape=(28, 28, 1)), 
        layers.Conv2D(32, (3, 3), activation="relu"),
        layers.MaxPooling2D((2, 2)),

        layers.Conv2D(64, (3, 3), activation="relu"),
        layers.MaxPooling2D((2, 2)),

        layers.Flatten(),
    layers.Dense(128, activation="relu"),
        layers.Dense(10, activation="softmax"),
    ]
)


model_base.summary()

# -------------------------------------------------------------------
# 2. Compilar el modelo
# -------------------------------------------------------------------
model_base.compile(
    optimizer="adam",
    loss="categorical_crossentropy",
    metrics=["accuracy"],
)

# -------------------------------------------------------------------
# 3. Entrenar el modelo
# -------------------------------------------------------------------
EPOCHS = 10
BATCH_SIZE = 128

history_base = model_base.fit(
    x_train,
    y_train_cat,
    epochs=EPOCHS,
    batch_size=BATCH_SIZE,
    validation_data=(x_val, y_val_cat),
    verbose=1,
)

# -------------------------------------------------------------------
# 4. Graficar curvas de entrenamiento (accuracy y loss)
# -------------------------------------------------------------------
def plot_training_history(history):
    # Accuracy
    plt.figure(figsize=(8, 4))
    plt.plot(history.history["accuracy"], label="Entrenamiento")
    plt.plot(history.history["val_accuracy"], label="Validacin")
    plt.title("Exactitud (accuracy) durante el entrenamiento")
    plt.xlabel("poca")
    plt.ylabel("Accuracy")
    plt.legend()
    plt.grid(True)
    plt.show()

    # Loss
    plt.figure(figsize=(8, 4))
    plt.plot(history.history["loss"], label="Entrenamiento")
    plt.plot(history.history["val_loss"], label="Validacin")
    plt.title("Prdida (loss) durante el entrenamiento")
    plt.xlabel("poca")
    plt.ylabel("Loss")
    plt.legend()
    plt.grid(True)
    plt.show()

plot_training_history(history_base)

# -------------------------------------------------------------------
# 5. Imprimir resultados finales de entrenamiento y validacin
# -------------------------------------------------------------------
final_train_acc = history_base.history["accuracy"][-1]
final_val_acc = history_base.history["val_accuracy"][-1]
final_train_loss = history_base.history["loss"][-1]
final_val_loss = history_base.history["val_loss"][-1]

print("\nResultados finales del modelo base:")
print(f"Accuracy entrenamiento: {final_train_acc:.4f}")
print(f"Accuracy validacin   : {final_val_acc:.4f}")
print(f"Loss entrenamiento    : {final_train_loss:.4f}")
print(f"Loss validacin       : {final_val_loss:.4f}")

# -------------------------------------------------------------------
# 6. Guardar el modelo base para usos posteriores
# -------------------------------------------------------------------
model_base.save("modelo_mnist_base.keras")
print("\nModelo base guardado como 'modelo_mnist_base.keras'")

Paso 3 Seleccin y preprocesamiento de 50 imgenes externas (280280)

En este paso vamos a usar las 100 imgenes externas que generaste (10 por cada dgito del 0 al 9).
Estas imgenes son de 280280 pixeles, con fondo blanco y dgito en negro.

El objetivo es:

  • Seleccionar 5 imgenes por cada dgito (50 en total).
  • Preprocesarlas para que sean compatibles con el modelo entrenado con MNIST:
    • Convertir a escala de grises.
    • Aplicar binarizacin con inversin (THRESH_BINARY_INV), de modo que el dgito quede blanco sobre fondo negro, igual que en MNIST.
    • Redimensionar de 280280 a 2828.
    • Normalizar los pixeles al rango [0, 1].
    • Aadir el canal para que queden de tamao (28, 28, 1).

Al final obtendremos dos arreglos:

  • X_50: las 50 imgenes preprocesadas listas para el modelo.
  • y_50: las etiquetas verdaderas correspondientes (09, cinco veces cada una).

# Paso 3  Seleccin y preprocesamiento de 50 imgenes externas

import os
import cv2
import numpy as np
import matplotlib.pyplot as plt

# -------------------------------------------------------------------
# 1. Definir carpeta base de las imgenes
# -------------------------------------------------------------------
# Descomprime numeros.zip en el mismo directorio de tu notebook/script.
# El ZIP que compartiste genera una carpeta llamada "q" con subcarpetas
# "0", "1", ..., "9", cada una con ~10 imgenes PNG.
#
# Estructura esperada:
#   q/
#     0/611471_0_1.png, ...
#     1/611471_1_1.png, ...
#     ...
#     9/...
#
BASE_DIR = "numeros"

if not os.path.isdir(BASE_DIR):
    raise FileNotFoundError(
        f"No se encontr la carpeta '{BASE_DIR}'. "
        "Asegrate de haber descomprimido numeros.zip en el directorio actual."
    )

print(f"Usando carpeta base de imgenes externas: {BASE_DIR}")

# -------------------------------------------------------------------
# 2. Funcin de preprocesamiento (como en el taller)
#    - Lee en escala de grises
#    - Umbraliza e invierte (dgito blanco, fondo negro)
#    - Redimensiona a 2828
#    - Normaliza a [0, 1]
#    - Agrega canal para usar con la CNN (28281)
# -------------------------------------------------------------------
def preprocess_external_digit(path, threshold_value=150, debug=False):
    # Leer imagen en escala de grises
    img = cv2.imread(path, cv2.IMREAD_GRAYSCALE)
    if img is None:
        raise ValueError(f"No se pudo leer la imagen: {path}")

    # Binarizacin con inversin (dgito blanco sobre fondo negro)
    _, thresh = cv2.threshold(
        img, threshold_value, 255, cv2.THRESH_BINARY_INV
    )

    # Redimensionar a 28x28
    resized = cv2.resize(thresh, (28, 28), interpolation=cv2.INTER_AREA)

    # Normalizar a [0, 1] y agregar canal
    norm = resized.astype("float32") / 255.0   # 0.0 o 1.0
    norm = np.expand_dims(norm, axis=-1)       # (28, 28) -> (28, 28, 1)

    if debug:
        # Mostrar el flujo completo para inspeccin
        fig, axes = plt.subplots(1, 3, figsize=(9, 3))
        axes[0].imshow(img, cmap="gray")
        axes[0].set_title("Original 280280")
        axes[0].axis("off")

        axes[1].imshow(thresh, cmap="gray")
        axes[1].set_title("Umbralizada + invertida")
        axes[1].axis("off")

        axes[2].imshow(resized, cmap="gray")
        axes[2].set_title("Redimensionada 2828")
        axes[2].axis("off")

        plt.tight_layout()
        plt.show()

    return norm

# 3. Recorrer dgitos 09 y tomar 5 imgenes por clase
# -------------------------------------------------------------------
X_50 = []
y_50 = []

for digit in range(10):
    digit_dir = os.path.join(BASE_DIR, str(digit))
    if not os.path.isdir(digit_dir):
        raise FileNotFoundError(
            f"No se encontr la carpeta para el dgito {digit}: {digit_dir}"
        )

    # Listar imgenes .png del dgito
    files = [
        f for f in os.listdir(digit_dir)
        if f.lower().endswith(".png")
    ]
    files = sorted(files)  # para que sea determinista

    if len(files) < 5:
        raise ValueError(
            f"La carpeta {digit_dir} solo tiene {len(files)} imgenes, "
            "se necesitan al menos 5 por dgito."
        )

    # Seleccionar las primeras 5 imgenes (podras usar random.sample si quieres aleatorio)
    selected_files = files[:5]

    print(f"Dgito {digit}: usando archivos -> {selected_files}")

    for fname in selected_files:
        path = os.path.join(digit_dir, fname)
        processed_img = preprocess_external_digit(path)
        X_50.append(processed_img)
        y_50.append(digit)

Anlisis del Paso 4 Evaluacin con imgenes externas

Al evaluar el modelo con las 50 imgenes externas (5 por cada dgito del 0 al 9), se obtuvo una exactitud de:

  • Accuracy externo: 0.56
  • Errores totales: 22 de 50 imgenes

Este desempeo es considerablemente menor al obtenido en los conjuntos de validacin y prueba de MNIST, donde el modelo alcanz aproximadamente 0.988 de exactitud. La diferencia se explica por las caractersticas particulares de las imgenes externas y por cmo estas difieren de las imgenes en MNIST.

Principales observaciones sobre los errores

La mayora de los errores se concentran en ciertos patrones:

  1. Confusin de 0 3
    Varios ceros fueron clasificados como dgitos 3 debido a:

    • Trazos muy delgados.
    • Forma circular incompleta o irregular.
    • Ausencia de bordes fuertes como los que aparecen en MNIST.

    El modelo aprendi a interpretar dgitos con estilo manuscrito ms grueso y contrastado, por lo que los ceros del dataset externo son demasiado abiertos y similares al inicio del trazo de un 3.

  2. Confusin de 4 2 o 3
    Algunos cuatros presentan trazos rectos muy finos y sin el tpico ngulo pronunciado del 4 manuscrito de MNIST, lo que hace que el modelo los interprete como curvas parciales correspondientes al 2 o al 3.

  3. Confusin de 5 3
    Los cincos tienen trazos suaves y redondeados, sin el giro fuerte tpico del 5 de MNIST. Esto los hace visualmente parecidos a un 3 incompleto para la CNN entrenada.

  4. Trazos muy delgados y bordes suaves
    Las imgenes externas tienen mucha suavidad en los contornos debido al escalado y a la forma en que fueron generadas.
    Esto causa que los dgitos pierdan caractersticas de borde que son esenciales para la red.

  5. Variabilidad distinta a MNIST
    El modelo est acostumbrado a:

    • Dgitos ms gruesos
    • Contraste ms alto
    • Estilos especficos del dataset MNIST

    Las imgenes externas no siguen ese patrn, y por eso el modelo muestra dificultad para generalizar.

Conclusin del desempeo externo

El modelo generaliza bien dentro de MNIST, pero sufre cuando se expone a un estilo de escritura diferente.
La precisin externa del 56% revela que:

  • El preprocesamiento base es correcto (invertir, binarizar, redimensionar).
  • Sin embargo, se requiere adaptar el modelo o los datos para manejar imgenes con trazos ms delgados, bordes ms suaves y formas menos consistentes.

Este resultado es til para el siguiente paso, donde se propondrn e implementarn mejoras al modelo y al preprocesamiento para mejorar el rendimiento en imgenes externas.

Paso 5 Implementacin de tres mejoras al sistema

En este paso se implementan tres mejoras para intentar aumentar la capacidad de generalizacin del modelo, especialmente sobre las imgenes externas:

  1. Data augmentation: se aplican transformaciones aleatorias (rotaciones, zoom y traslaciones pequeas) a las imgenes de entrenamiento, para que el modelo vea dgitos con variaciones de posicin y forma.

  2. Mejora de la arquitectura: se define un nuevo modelo convolucional que incluye capas de BatchNormalization y Dropout.

    • BatchNormalization estabiliza el entrenamiento.
    • Dropout ayuda a reducir el sobreajuste.

  3. Preprocesamiento mejorado de imgenes externas: se modifica el preprocesamiento para aplicar una operacin de dilatacin morfolgica sobre los dgitos, de manera que los trazos se vuelvan ms gruesos y se parezcan ms al estilo de los dgitos de MNIST.

Despus de implementar estas mejoras:

  • Se vuelve a entrenar el modelo usando las mismas particiones de entrenamiento y validacin.
  • Se evala el nuevo modelo en el conjunto de prueba.
  • Se mide el desempeo sobre las 50 imgenes externas usando el nuevo preprocesamiento y se compara con el modelo base.

Paso 6 Sistema de prediccin en tiempo real con cmara

En este paso se implementa un sistema sencillo que usa la cmara web para reconocer dgitos escritos a mano.
El usuario coloca un dgito dentro de un recuadro en la imagen, el sistema:

  1. Recorta esa regin (ROI).
  2. La convierte a escala de grises.
  3. Aplica binarizacin + inversin + dilatacin para engrosar el trazo.
  4. Redimensiona a 2828 y normaliza.
  5. Usa el modelo entrenado para predecir el dgito.
  6. Muestra en pantalla la cmara con el recuadro y el dgito predicho.

Se utiliza el modelo base porque, aun con sus limitaciones, es el que mejor funcion sobre las imgenes externas con preprocesamiento dilatado.

kernel = np.ones((3, 3), np.uint8)

def preprocess_roi(roi, threshold_value=150):
    # ROI viene en BGR
    gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
    _, thresh = cv2.threshold(gray, threshold_value, 255, cv2.THRESH_BINARY_INV)
    dilated = cv2.dilate(thresh, kernel, iterations=1)
    resized = cv2.resize(dilated, (28, 28), interpolation=cv2.INTER_AREA)

    norm = resized.astype("float32") / 255.0
    norm = np.expand_dims(norm, axis=-1)   # (28, 28, 1)
    norm = np.expand_dims(norm, axis=0)    # (1, 28, 28, 1)
    return norm
